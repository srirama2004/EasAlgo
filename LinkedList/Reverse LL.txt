class Solution {
    public ListNode reverseList(ListNode head) {
        if(head==null || head.next==null){
            return head;
        }
        ListNode r=null;
        ListNode p=head.next;
        ListNode q=head;
        while(q.next!=null){
            q.next=r;
            r=q;
            q=p;
            p=p.next;
        }
        q.next=r;
        head=q;
       return head;
    }
}
